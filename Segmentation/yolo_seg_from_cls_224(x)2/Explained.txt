def preprocess_image(img):
    img = cv2.resize(img, (224, 224))
    # img = img.astype(np.float32)     
    return img

def preprocess_mask(mask, target_size=(224, 224)):
    mask = cv2.resize(mask, target_size, interpolation=cv2.INTER_NEAREST)
    return mask
